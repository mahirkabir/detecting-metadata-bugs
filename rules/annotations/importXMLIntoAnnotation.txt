Rule import-resource-path {
	for (class c in getClasses()) {
		for(String s in getAnnoAttr(c, "@ImportResource", "locations")) {
			assert(pathExists(s)) {
				msg ("The @ImportResource path: %s in class: %s does not exist", s, getFQN(c));
			}
		}
		for(String s in getAnnoAttr(c, "@ImportResource", "value")) {
			assert(pathExists(s)) {
				msg ("The @ImportResource path: %s in class: %s does not exist", s, getFQN(c));
			}
		}
	}
}
