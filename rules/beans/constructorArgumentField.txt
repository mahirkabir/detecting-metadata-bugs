Rule constructor-arg-name-field-map {
	for (class c in getClasses()) {
		for (file xml in getXMLs()) {
			for (<bean> bean in getElms(xml, "<bean>")) {
				if (getAttr(bean, "class") == getFQN(c)) {
					for (<constructor-arg> constructor_arg in getElms(bean, "<constructor-arg>")) {
						if (NOT isEmpty(getAttr(constructor_arg, "name"))) {
							assert(exists(method con in getConstructors(c))(hasParam(con, getAttr(constructor_arg, "name")))) {
								msg ("The name of <constructor-arg>: %s in bean: %s does not correspond to any constuctor-parameter's name in class: %s", getAttr(constructor_arg, "name"), getName(bean), getFQN(c));
							}
						}
					}
				}
			}
		}
	}
}
