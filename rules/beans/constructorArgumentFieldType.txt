Rule constructor-arg-type-field-map {
	for (class c in getClasses()) {
		for (file xml in getXMLs()) {
			for (<bean> bean in getElms(xml, "<bean>")) { // @bean
				if (getAttr(bean, "class") == getFQN(c)) {
					for (<constructor-arg> constructor_arg in getElms(bean, "<constructor-arg>")) {
						if (NOT isEmpty(getAttr(constructor_arg, "type"))) {
							assert(exists(method con in getConstructors(c))(hasParamType(con, getAttr(constructor_arg, "type")))) {
								msg ("The type of <constructor-arg>: %s does not correspond to any constuctor-parameter's type in class: %s", getAttr(constructor_arg, "type"), getFQN(c));
							}
						}
					}
				}
			}
		}
  }
}
